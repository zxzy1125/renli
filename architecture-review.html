// 新建 src/hooks/useListPage.ts
export function useListPage<T>({
  fetchApi,
  defaultPageSize = 12,
}: UseListPageOptions<T>) {
  const [list, setList] = useState<T[]>([]);
  const [total, setTotal] = useState(0);
  const [page, setPage] = useState(1);
  const [keyword, setKeyword] = useState('');
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState('');

  const fetchList = useCallback(async () => {
    setLoading(true);
    try {
      const res = await fetchApi({ keyword, page, pageSize: defaultPageSize });
      setList(res.data);
      setTotal(res.total);
    } catch (err) {
      setError(getErrorMsg(err));
    } finally {
      setLoading(false);
    }
  }, [keyword, page, fetchApi]);

  return { list, total, page, setPage, keyword, setKeyword, loading, error, fetchList };
}

收益

  • 局部性：列表逻辑集中在一个地方，修改一处即可影响所有列表页面
  • 杠杆性：新增列表页面只需几行代码调用 Hook
  • 可测试性：Hook 可以独立进行单元测试
  • 代码减少：每个列表页面减少约 150-200 行重复代码

// 新建 server/src/repositories/base.ts
export function buildUpdateSQL<T>(
  table: string,
  id: string,
  input: Partial<T>,
  fieldMap: Record<string, (val: any) => any>
): void {
  const fields: string[] = [];
  const values: any[] = [];
  
  for (const [key, transform] of Object.entries(fieldMap)) {
    if (input[key] !== undefined) {
      fields.push(`${key} = ?`);
      values.push(transform(input[key]));
    }
  }
  
  if (fields.length === 0) return;
  fields.push("updated_at = datetime('now')");
  values.push(id);
  
  db.prepare(`UPDATE ${table} SET ${fields.join(', ')} WHERE id = ?`)
    .run(...values);
}

// 使用示例 - positionRepo.ts 简化为：
export function updatePosition(id: string, input: UpdatePositionInput) {
  buildUpdateSQL('positions', id, input, {
    title: String,
    client_id: v => v ?? null,
    keywords: v => v ? JSON.stringify(v) : null,
    // ... 其他字段
  });
  return findPositionById(id);
}

收益

  • 删除测试：删除 base.ts 会使所有 Repo 崩塌，说明它是真正的深度模块
  • 杠杆性：新增实体的 Repository 只需定义字段映射
  • 代码减少：每个 Repository 减少约 100-150 行

问题

  • POST /positions 有 30+ 行参数映射代码
  • 手动类型转换（String(), Number()）
  • 缺少输入验证（如字段长度、格式）

解决方案

使用 zod 进行 schema 验证（已在 package.json 中）：

// server/src/schemas/position.ts
import { z } from 'zod';

export const createPositionSchema = z.object({
  title: z.string().min(1, '职位标题不能为空').max(100),
  client_id: z.string().nullable().optional(),
  department: z.string().max(50).nullable().optional(),
  salary_min: z.string().max(20).nullable().optional(),
  // ... 其他字段
  keywords: z.array(z.string()).default([]),
});

// 路由简化为：
positionsRouter.post('/', requireAdmin, asyncHandler(async (req, res) => {
  const input = createPositionSchema.parse(req.body);
  const p = createPosition({ ...input, id: nanoid(), created_by: req.user!.id });
  res.status(201).json({ data: p });
}));

收益

  • 安全性：自动验证所有输入，防止注入攻击
  • 代码减少：路由代码减少约 60%
  • 类型安全：zod 推导 TypeScript 类型

  • 页面切换时数据重新加载
  • 没有请求去重和缓存机制

解决方案

引入 React Query 或 SWR 进行数据缓存：

// 使用 React Query 示例
import { useQuery } from '@tanstack/react-query';

export function useClients() {
  return useQuery({
    queryKey: ['clients'],
    queryFn: () => clientsApi.list(),
    staleTime: 5 * 60 * 1000, // 5分钟缓存
  });
}

收益

  • 性能：减少不必要的 API 调用
  • 用户体验：页面切换更快，数据即时显示
  • 代码简化：不需要手动管理 loading/error 状态
